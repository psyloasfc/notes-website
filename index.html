Experiment 2: Linear Convolution

clc; clear; close all; x = [1 2 3 4]; % Input signal x(n) h = [1 1 1]; % Impulse response h(n) Nx = length(x); Nh = length(h); Ny = Nx + Nh - 1; y_manual = zeros(1, Ny); for n = 1:Ny for k = 1:Nx if (n - k + 1 > 0) && (n - k + 1 <= Nh) y_manual(n) = y_manual(n) + x(k) * h(n - k + 1); end end end y_builtin = conv(x, h); disp('Manual Convolution Output:'); disp(y_manual); disp('MATLAB conv() Output:'); disp(y_builtin); % Plot signals n_x = 0:Nx-1; n_h = 0:Nh-1; n_y = 0:Ny-1; figure; subplot(3,1,1); stem(n_x, x, 'filled'); title('Input Signal x(n)'); xlabel('n'); ylabel('Amplitude'); grid on; subplot(3,1,2); stem(n_h, h, 'filled'); title('Impulse Response h(n)'); xlabel('n'); ylabel('Amplitude'); grid on; subplot(3,1,3); stem(n_y, y_builtin, 'filled'); title('Output Signal y(n) = x(n) * h(n)'); xlabel('n'); ylabel('Amplitude'); grid on;

Experiment 5: DFT vs DCT

clc; clear; close all; % Read Image img = imread('cameraman.tif'); img = im2double(img); figure; imshow(img); title('Original Image'); %% ----------- DFT ----------- F = fft2(img); F_shift = fftshift(F); magnitude_spectrum = log(1 + abs(F_shift)); figure; imshow(magnitude_spectrum,[]); title('DFT Magnitude Spectrum'); % Inverse DFT img_dft_recon = real(ifft2(F)); figure; imshow(img_dft_recon); title('Reconstructed Image using IDFT'); %% ----------- DCT ----------- C = dct2(img); figure; imshow(log(1+abs(C)),[]); title('DCT Coefficients'); % Inverse DCT img_dct_recon = idct2(C); figure; imshow(img_dct_recon); title('Reconstructed Image using IDCT'); %% Energy Comparison dft_energy = sum(abs(F(:)).^2); dct_energy = sum(abs(C(:)).^2); disp(['Total DFT Energy: ', num2str(dft_energy)]); disp(['Total DCT Energy: ', num2str(dct_energy)]);
